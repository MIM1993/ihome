# 容器化微服

1、编译项目

```shell
 CGO_ENABLED=0 GOOS=linux  /home/itcast/go/bin/go build -a -installsuffix cgo -ldflags '-w' -i -o 服务名 main.go
```

2、编写Dockerfile文件

web

```Dockerfile
FROM alpine:3.2
#拷贝文件
ADD conf /conf
#拷贝文件
ADD html /html
#拷贝二进制
ADD ihomeweb /ihomeweb
WORKDIR /

ENTRYPOINT [ "/ihomeweb" ]

EXPOSE 23333
```

srv

```Dockerfile
FROM alpine:3.2
ADD conf /conf

ADD getarea-srv /getarea-srv
ENTRYPOINT [ "/getarea-srv" ]
```



3、编译dockers-compose.yml文件

```yml
consul:
  #覆盖启动后的执行命令
#  command: agent -server -bootstrap-expect=1  -node=node1 -client 0.0.0.0 -ui -bind=0.0.0.0 -join 127.0.0.2
  command: agent -server -bootstrap-expect=1 -advertise-wan=192.168.11.145 -node=node1 -client 0.0.0.0 -ui -bind=0.0.0.0 -join 192.168.11.145
  #-advertise-wan=192.168.11.145  表示公网或者大部分网络可以到达的地方
  #镜像：镜像名称:版本号
  image: consul:latest
  #主机名
  hostname: "registry"
  #暴露端口
  ports:
  - "8300:8300"
  - "8301:8301"
  - "8400:8400"
  - "8500:8500"
  - "8600:53/udp"

#nginx容器化搭建
n1:
  image: nginx
  ports:
    - "12580:80"
  volumes:
    - /home/itcast/workspace/go/src/sss/nginx/nginx.conf:/etc/nginx/conf.d/default.conf
    #将配置文件映射到nginx的配置文件位置
  container_name: nginx1


#web1主页
web1:
  #覆盖启动后的执行命令
  command: --registry_address=192.168.11.145:8500 --register_interval=5 --register_ttl=10 web
  #镜像构建的dockerfile文件地址
  build: ./IhomeWeb
  links:
  - consul
  ports:
  - "12306:23333"

#web1主页
web2:
  #覆盖启动后的执行命令
  command: --registry_address=192.168.11.145:8500 --register_interval=5 --register_ttl=10 web
  #镜像构建的dockerfile文件地址
  build: ./IhomeWeb
  links:
  - consul
  ports:
  - "12307:23333"

#获取地区
getarea:
  #覆盖启动后的执行命令
  command: --registry_address=192.168.11.145:8500 --register_interval=5 --register_ttl=10 srv
  #镜像构建的dockerfile文件地址
  build: ./GetArea
  links:
  - consul

#登录服务，名字不能大写
postlogin:
  #覆盖启动后的执行命令
  command: --registry_address=192.168.11.145:8500 --register_interval=5 --register_ttl=10 srv
  #镜像构建的dockerfile文件地址
  build: ./PostLogin
  links:
   - consul

#退出登录服务
deletesession:
  #覆盖启动后的执行命令
  command: --registry_address=192.168.11.145:8500 --register_interval=5 --register_ttl=10 srv
  #镜像构建的dockerfile文件地址
  build: ./DeleteSession
  links:
   - consul

```



编译docker-compose.tml文件

```shell
docker-compose build --no-cache
```

启动命令：

```shell
docker-compose up
```
